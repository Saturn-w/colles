\subsection{Démonstration de : \\ 
si $R_a$ le rayon de convergence de $\sum a_n z^n$ et $R_b$ celui de $\sum b_n z^n$ \\ 
alors si $|a_n| \le |b_n|$ alors $R_a \ge R_b$}

\color{black}
\vspace{0.5cm}

\noindent Soit $r \ge 0$ tel que $r < R_b$ \\
\noindent alors $(b_n r^n)_n$ est bornée :

\vspace{0.3cm}

\noindent $\bullet$ $\exists M \ge 0, \forall n \in \mathbb{N}, \enspace |b_n r^n| \le M$

\vspace{0.3cm}

\noindent $\rightarrow$ Donc $\forall n \in \mathbb{N}, \enspace |a_n r^n| \le M$ \\
\noindent $\rightarrow$ Donc $(a_n r^n)$ est bornée

\vspace{0.3cm}

\noindent Donc $[0, R_b[ \enspace \subset [0, R_a]$ \\
\noindent Donc $R_b \le R_a$

\vspace{0.5cm}

\noindent D'où :
\[ \fbox{$R_a \ge R_b$} \]
