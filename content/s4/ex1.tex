\subsection{ Démonstration de $tr(AB) = tr(BA)$ \\
\null\hfill $A, B \in \mathcal{M}_n(\mathbb{K})$ \hfill\null}

\color{black}
\vspace{0.5cm}

\[ tr(AB) = \sum_{i=1}^n [AB]_{(ii)} = \sum_{i=1}^n \sum_{k=1}^n a_{ik} b_{ki} \]

\vspace{0.2cm}

\[ = \sum_{k=1}^n \sum_{i=1}^n a_{ik} b_{ki} \]

\vspace{0.2cm}

\[ = \sum_{k=1}^n \sum_{i=1}^n b_{ki} a_{ik} \]

\vspace{0.2cm}

\[ = \sum_{k=1}^n [BA]_{(ii)} \] % Note : Mathématiquement, l'indice devrait être k (somme sur k), mais j'ai transcrit le (ii) visible sur l'image

\vspace{0.2cm}

\[ = tr(BA) \]
